```typescript
export interface QuantityStats {
  uniqueOwned: number;      // Count of cardIds with qty > 0
  totalQuantity: number;    // Sum of all quantities
  missing: number;          // total cards - uniqueOwned
  completionPercent: number; // (uniqueOwned / totalCards) * 100
}

export function computeQuantityStats(
  cardIds: string[],
  cardQuantities: Record<string, number>
): QuantityStats
```

Implementation: 
- `uniqueOwned = cardIds.filter(id => (cardQuantities[id] ?? 0) > 0).length`
- `totalQuantity = cardIds.reduce((sum, id) => sum + (cardQuantities[id] ?? 0), 0)`
- `missing = cardIds.length - uniqueOwned`
- `completionPercent = cardIds.length > 0 ? (uniqueOwned / cardIds.length) * 100 : 0`
