1. Import types from `./collection-types`
2. Export `createBackup(data: CollectionStateV1): boolean` - writes to BACKUP_KEY, returns success
3. Export `restoreFromBackup(): CollectionStateV1 | null` - reads from BACKUP_KEY
4. Export `clearBackup(): void` - removes BACKUP_KEY
5. Export `migrateV1ToV3(v1State: CollectionStateV1): CollectionStateV3`:
   ```typescript
   export const migrateV1ToV3 = (v1State: CollectionStateV1): CollectionStateV3 => {
     // Backup first
     createBackup(v1State);
     
     try {
       // Transform: true → 1, false/undefined → omit (sparse)
       const cardQuantities: Record<string, number> = {};
       Object.entries(v1State.ownedCards).forEach(([id, owned]) => {
         if (owned === true) {
           cardQuantities[id] = 1;
         }
         // Sparse: omit false/falsy entries
       });
       
       const v3State: CollectionStateV3 = {
         version: 3,
         cardQuantities
       };
       
       // Validate with Zod
       CollectionV3Schema.parse(v3State);
       
       console.log(`Migration complete: ${Object.keys(cardQuantities).length} cards`);
       return v3State;
       
     } catch (e) {
       console.error('Migration failed:', e);
       // Attempt rollback
       const backup = restoreFromBackup();
       if (backup) {
         console.warn('Restored from backup, retrying in-memory...');
         // Return in-memory conversion (don't persist failed state)
         return convertToV3InMemory(backup);
       }
       // Last resort: empty state
       console.error('Rollback failed, using empty collection');
       return { version: 3, cardQuantities: {} };
     }
   };
   ```
6. Export helper `getInitialState(): CollectionStateV3`:

   - If unknown version → log warning, return default
